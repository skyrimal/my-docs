# tomcat部署项目后修改访问路径

```xml
	<Host name="localhost"  appBase="webapps"
            unpackWARs="true" autoDeploy="true">

        <!--添加Context标签，如下
			docBase指向项目路径
			path是修改后的映射路径
			reloadable开启功能-->

		<Context docBase="C:\Program Files\apache-tomcat-7.0.100\webapps\permission-0.0.1-SNAPSHOT"
		path="/gsweb"
		reloadable="true" />
      </Host>
```
