# sql过滤器

not EXISTS

```sql
SELECT a.taskid from prd_scgs_tmp.jihe_task a 
left join  prd_scgs_tmp.jihe_taskproc b on a.taskid=b.taskid
where b.tasktype='taskstatus' and 
 not EXISTS (
 select procid from prd_scgs_tmp.jihe_taskproc  c where b.procid=c.procid and c.tasktype='taskstatus' and c.dictid  in ('000007')
 ) 
ORDER BY proctime desc
```
